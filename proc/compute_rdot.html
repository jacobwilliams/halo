    function compute_rdot(rv) result(rdot)
        real(wp),dimension(6),intent(in) :: rv !! rv vector (km, km/s)
        real(wp) :: rdot !! [km/s]
        associate(r => rv(1:3), v => rv(4:6))
            rdot = dot_product(r,v) / norm2(r)
        end associate
    end function compute_rdot
