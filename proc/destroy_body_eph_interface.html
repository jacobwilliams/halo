    subroutine destroy_body_eph_interface(me)
        class(body_eph_interface),intent(inout) :: me
        if (associated(me%eph)) call me%eph%destroy()
        me%eph => null()
        if (allocated(me%w0)) deallocate(me%w0)
        me%inbvx = 0
    end subroutine destroy_body_eph_interface
